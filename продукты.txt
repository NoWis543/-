            const question = qNode ? qNode.innerText.trim() : null;

            if (question && question !== prevQuestion) {

                let checkRows = setInterval( () => {
                    const rows = document.querySelectorAll('[data-testid="Варианты"] [data-component="list-row"]');

                        // нашли варианты, дальше обрабатываем

                                const chosen = Array.from(rows).find(r => {

                                    return lbl && lbl.innerText.trim() === expectedAnswer;
                                }
                                );

                                    chosen.scrollIntoView({

                                    });
